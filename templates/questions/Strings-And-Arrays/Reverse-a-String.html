Question

Given a string of length n, return a new string such that the ith character of the new string is equal to the n-ith character in the original string

def reverse_string(string):
  length_of_string = len(string)
  char_list = [string[i] for i in xrange(length_of_string - 1, -1, -1)]
  return ''.join(char_list)
	
function reverseString(originalString){
  var reversedString = [];
  var string_length = originalString.length;
  for (var i = 0; i < string_length; i++){
    reversedString[i] = originalString[string_length - i - 1];
  }
  return reversedString.join('');
}
